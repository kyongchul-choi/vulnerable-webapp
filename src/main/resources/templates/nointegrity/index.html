무결성 검사 없는 코드 다운로드

설명: 무결성 검사 없이 코드 또는 바이너리를 다운로드하는 것은 중간자 공격(Man-in-the-Middle)이나 악의적인 코드 삽입을 허용할 수 있는 취약점을 초래합니다.

대책: SHA256 해시값 또는 디지털 서명과 같은 무결성 검사를 통해 다운로드된 파일이 변조되지 않았는지 확인하세요.

샘플 코드

취약 코드:

import java.io.*;

import java.net.*;

public class FileDownloader {

public void downloadFile(String url, String outputPath) throws IOException {

URL website = new URL(url);

try (InputStream in = website.openStream();

FileOutputStream fos = new FileOutputStream(outputPath)) {

byte[] buffer = new byte[1024];

int bytesRead;

while ((bytesRead = in.read(buffer)) != -1) {

fos.write(buffer, 0, bytesRead);

}

}

}

}

안전 코드:

import java.io.*;

import java.net.*;

import java.security.*;

; import java.nio.file.*;

public class SecureFileDownloader {

public void downloadFileWithIntegrityCheck(String url, String outputPath, String expectedHash) throws Exception {

URL website = new URL(url);

MessageDigest md = MessageDigest.getInstance("SHA-256");

try (InputStream in = website.openStream();

FileOutputStream fos = new FileOutputStream(outputPath)) {

byte[] buffer = new byte[1024];

int bytesRead;

while ((bytesRead = in.read(buffer)) != -1) {

fos.write(buffer, 0, bytesRead);

md.update(buffer, 0, bytesRead);

}

}

String downloadedHash = bytesToHex(md.digest());

if (!downloadedHash.equals(expectedHash)) {

throw new SecurityException("무결성 검사 실패: 파일이 변조되었습니다.");

}

}

private String bytesToHex(byte[] bytes) {

StringBuilder sb = new StringBuilder();

for (byte b : bytes) {

sb.append(String.format("%02x", b));

}

return sb.toString();

}

}
